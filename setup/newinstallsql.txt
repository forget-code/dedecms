CREATE TABLE dede_admin (
  ID int(10) unsigned NOT NULL auto_increment,
  usertype int(10) unsigned NOT NULL default '0',
  userid varchar(30) NOT NULL default '',
  pwd varchar(50) NOT NULL default '',
  uname varchar(20) NOT NULL default '',
  typeid int(10) unsigned NOT NULL default '0',
  logintime datetime NOT NULL default '0000-00-00 00:00:00',
  loginip varchar(20) NOT NULL default '',
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_aera (
  ID int(10) unsigned NOT NULL default '0',
  name varchar(10) NOT NULL default ''
);
CREATE TABLE dede_art (
  ID int(10) unsigned NOT NULL auto_increment,
  typeid int(10) unsigned NOT NULL default '0',
  isdd smallint(5) unsigned NOT NULL default '0',
  redtitle smallint(5) unsigned NOT NULL default '0',
  ismake smallint(5) unsigned NOT NULL default '0',
  rank smallint(5) NOT NULL default '0',
  click int(10) unsigned NOT NULL default '0',
  picname varchar(200) NOT NULL default '',
  title varchar(100) NOT NULL default '',
  writer varchar(50) NOT NULL default '',
  source varchar(50) NOT NULL default '',
  msg text NOT NULL,
  stime date NOT NULL default '0000-00-00',
  likeid varchar(250) NOT NULL default '',
  body mediumtext NOT NULL,
  userid int(10) unsigned NOT NULL default '0',
  spec int(10) unsigned NOT NULL default '0',
  dtime datetime NOT NULL default '0000-00-00 00:00:00',
  memberID int(11) NOT NULL default '0',
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_arttype (
  ID int(10) unsigned NOT NULL auto_increment,
  reID int(10) unsigned NOT NULL default '0',
  typename varchar(30) NOT NULL default '',
  typedir varchar(100) NOT NULL default '',
  isdefault smallint(6) NOT NULL default '0',
  defaultname varchar(30) NOT NULL default 'index.html',
  issend smallint(6) NOT NULL default '0',
  channeltype smallint(6) NOT NULL default '1',
  modname varchar(50) NOT NULL default '默认模板',
  maxpage int(11) NOT NULL default '-1',
  ispart smallint(6) NOT NULL default '0',
  description varchar(250) NOT NULL DEFAULT '',
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_channeltype (
  ID smallint(6) NOT NULL default '0',
  typename varchar(50) NOT NULL default ''
);
CREATE TABLE dede_feedback (
  ID int(10) unsigned NOT NULL auto_increment,
  artID int(10) unsigned NOT NULL default '0',
  userid int(10) unsigned NOT NULL default '0',
  username varchar(20) NOT NULL default '',
  ip varchar(20) NOT NULL default '',
  msg text NOT NULL,
  ischeck smallint(6) NOT NULL default '0',
  dtime datetime NOT NULL default '0000-00-00 00:00:00',
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_flink (
  ID int(11) NOT NULL auto_increment,
  url varchar(100) NOT NULL default '',
  webname varchar(30) NOT NULL default '',
  msg varchar(250) NOT NULL default '',
  email varchar(100) NOT NULL default '',
  logo varchar(100) NOT NULL default '',
  dtime datetime NOT NULL default '0000-00-00 00:00:00',
  typeid int(11) NOT NULL default '0',
  ischeck smallint(6) NOT NULL default '1',
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_flinktype (
  ID int(11) NOT NULL auto_increment,
  typename varchar(50) NOT NULL default '',
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_member (
  ID int(10) unsigned NOT NULL auto_increment,
  userid varchar(32) NOT NULL default '',
  pwd varchar(32) NOT NULL default '',
  uname varchar(20) NOT NULL default '',
  sex smallint(5) unsigned NOT NULL default '1',
  age varchar(5) NOT NULL default '',
  birthday varchar(20) NOT NULL default '0000-00-00',
  weight varchar(10) NOT NULL default '',
  height varchar(10) NOT NULL default '',
  job varchar(10) NOT NULL default '',
  aera smallint(5) unsigned NOT NULL default '1',
  city varchar(20) NOT NULL default '',
  myinfo varchar(250) default NULL,
  mybb varchar(250) NOT NULL default '',
  tel varchar(30) default NULL,
  oicq varchar(15) default NULL,
  email varchar(50) NOT NULL default '',
  homepage varchar(50) default NULL,
  jointime datetime NOT NULL default '0000-00-00 00:00:00',
  joinip varchar(20) NOT NULL default '',
  logintime datetime NOT NULL default '0000-00-00 00:00:00',
  loginip varchar(20) NOT NULL default '',
  rank smallint(5) unsigned NOT NULL default '0',
  isup smallint(6) NOT NULL default '0',
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_membertype (
  ID int(10) unsigned NOT NULL auto_increment,
  rank smallint(10) NOT NULL default '0',
  membername varchar(20) NOT NULL default '',
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_partmode (
  ID int(10) unsigned NOT NULL auto_increment,
  typeid int(10) unsigned NOT NULL default '0',
  pname varchar(50) NOT NULL default '',
  fname varchar(50) NOT NULL default '',
  body text NOT NULL,
  PRIMARY KEY  (ID)
);
CREATE TABLE dede_spec (
  ID int(10) unsigned NOT NULL auto_increment,
  AID int(10) unsigned NOT NULL default '0',
  typeid int(10) unsigned NOT NULL default '0',
  spectitle varchar(100) NOT NULL default '',
  imgtitle varchar(50) NOT NULL default '',
  imglink varchar(100) NOT NULL default '',
  specimg varchar(100) NOT NULL default '',
  specmsg text NOT NULL,
  specartid text NOT NULL,
  speclikeid text NOT NULL,
  stime date NOT NULL default '0000-00-00',
  dtime datetime NOT NULL default '0000-00-00 00:00:00',
  userid int(11) NOT NULL default '0',
  PRIMARY KEY  (ID),
  KEY AID (AID)
);
CREATE TABLE dede_weburl (
  ID int(10) unsigned NOT NULL auto_increment,
  title varchar(80) NOT NULL default '',
  url varchar(80) NOT NULL default '',
  msg varchar(250) NOT NULL default '',
  PRIMARY KEY  (ID)
);
